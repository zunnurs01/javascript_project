document.write()


                    //html
                    <button type="button" class="btn btn-info" onclick="docwritexmp()">Click Me!</button>

                    //javascript
                    function docwritexmp() {
                        var varWindow = window.open("","_blank");
                        varWindow.document.write("<h1>Hello World!</h1>");
                        varWindow.document.close();
                    };

alert() & confirm()


                    //html
                    <button type="button" class="btn btn-primary" onclick="alertconfirmFunction()">Click Me!</button>

                    //javascript
                    function alertconfirmFunction() {
                        var confirmation = confirm("Confirm?");
                        if (confirmation) {
                            alert('You selected "Ok"');
                        } else {
                            alert('You selected "Cancel"');
                        }
                    };
